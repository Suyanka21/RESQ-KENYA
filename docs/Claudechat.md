// Initialize services
export const auth = getAuth(app);
export const firestore = getFirestore(app);
export const database = getDatabase(app);
export const functions = getFunctions(app);
export const storage = getStorage(app);

// Connect to emulators (development only)
if (__DEV__) {
  const EMULATOR_HOST = 'localhost'; // Use your computer's IP for physical devices

  connectFirestoreEmulator(firestore, EMULATOR_HOST, 8080);

  connectFunctionsEmulator(functions, EMULATOR_HOST, 5001);

  console.log('🔥 Connected to Firebase Emulators');
}

interface ButtonProps {
  title: string;
  onPress: () => void;
  variant?: 'primary' | 'secondary' | 'outline';
  size?: 'small' | 'medium' | 'large';
  loading?: boolean;
  disabled?: boolean;
  style?: ViewStyle;
}

export const Button: React.FC<ButtonProps> = ({
  title,
  onPress,
  variant = 'primary',
  size = 'medium',
  loading = false,
  disabled = false,
  style,
}) => {
  const buttonStyles = [
    styles.button,
    styles[variant],
    styles[size],
    disabled && styles.disabled,
    style,
  ];

  const textStyles = [
    styles.text,
    styles[`${variant}Text`],
    styles[`${size}Text`],
  ];

  return (
    <TouchableOpacity
      style={buttonStyles}
      onPress={onPress}
      disabled={disabled || loading}
      activeOpacity={0.8}
    >
      {loading ? (
        <ActivityIndicator color={variant === 'primary' ? Colors.charcoal[900] : Colors.voltage} />
      ) : (
        <Text style={textStyles}>{title}</Text>
      )}
    </TouchableOpacity>
  );
};

    const locationRef = ref(database, `activeRequests/${requestId}/providerLocation`);

      const location = snapshot.val();
      if (location) {

          latitude: location.latitude,

Step 5.1: Create Service Layer

typescript
import { collection, addDoc, doc, getDoc, updateDoc, query, where, getDocs } from 'firebase/firestore';
import { firestore } from '../firebase.config';
import { ref, set, onValue, off } from 'firebase/database';
import { database } from '../firebase.config';

export interface ServiceRequest {
  userId: string;
  serviceType: string;
  customerLocation: {
    latitude: number;
    longitude: number;
    address: string;
  };
  status: 'pending' | 'accepted' | 'enroute' | 'arrived' | 'completed';
  // ... other fields from your backend schema
}

// Create new service request
export async function createServiceRequest(data: Partial<ServiceRequest>): Promise<string> {
  try {
    const requestRef = await addDoc(collection(firestore, 'requests'), {
      ...data,
      createdAt: new Date(),
      status: 'pending',
    });
    
    return requestRef.id;
  } catch (error) {
    console.error('Error creating request:', error);
    throw error;
  }
}

// Subscribe to request updates
export function subscribeToRequest(
  requestId: string,
  callback: (request: ServiceRequest) => void
) {
  const requestRef = doc(firestore, 'requests', requestId);
  
  return onValue(requestRef, (snapshot) => {
    const data = snapshot.data();
    if (data) {
      callback(data as ServiceRequest);
    }
  });
}

// Subscribe to provider location (Realtime Database)
export function subscribeToProviderLocation(
  requestId: string,
  callback: (location: { latitude: number; longitude: number }) => void
) {
  const locationRef = ref(database, `activeRequests/${requestId}/providerLocation`);
  
  const listener = onValue(locationRef, (snapshot) => {
    const location = snapshot.val();
    if (location) {
      callback({
        latitude: location.latitude,
        longitude: location.longitude,
      });
    }
  });

  return () => off(locationRef, 'value', listener);
}

// Get user's request history
export async function getRequestHistory(userId: string): Promise<ServiceRequest[]> {
  try {
    const requestsQuery = query(
      collection(firestore, 'requests'),
      where('userId', '==', userId)
    );
    
    const snapshot = await getDocs(requestsQuery);
    return snapshot.docs.map(doc => ({
      id: doc.id,
      ...doc.data(),
    })) as ServiceRequest[];
  } catch (error) {
    console.error('Error fetching history:', error);
    throw error;
  }
}
